pipeline {

       agent any
       
           stages{

            
                    

            stage("Test,Build"){
               steps{

                   bat "mvn clean install"
                    }

                  }

              stage("package"){
               steps{

                   bat "mvn package"
                    }

                  }
                  
               stage("Sonar"){
               steps{

                   bat "mvn sonar:sonar"
                    }

                  }
                  
                stage("Nexus"){
               steps{

                   bat "mvn clean package -Dmaven.test.failure.ignore=true deploy:deploy-file -DgroupId=tn.esprit.spring -DartifactId=Timesheet-spring-boot-core-data-jpa-mvc-REST-1 -Dversion=1.0 -DgeneratePom=true -Dpackaging=jar -DrepositoryId=deploymentRepo -Durl=http://localhost:8081/repository/maven-releases/ -Dfile=target/Timesheet-spring-boot-core-data-jpa-mvc-REST-1-1.2.jar"
                    }

                  }
           }
           
           
              stage("docker,build"){
               steps{

                   bat "FROM openjdk:8-jdk-alpine EXPOSE 8083 ADD target/docker-spring-boot.war docker-spring-boot.war  ENTRYPOINT["java","-jar","/docker-spring-boot.war"]"
                    }

                  }
           
           
           
           
           
           
}
